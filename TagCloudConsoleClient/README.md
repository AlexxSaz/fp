# Основные команды

```
help
image -w "ширина изображения" 
      -h "высота изображения" 
      -f "номер шрифта (от 1 до 6)" 
      -u "максимальный размер шрифта" 
      -l "минимальный размер шрифта"
logic -r "приращение радиуса" 
      -a "приращение угла"
      -g "форма облака (1 или 2)"
color -b "номер цвета фона (от 1 до 8)" 
      -f "номер цвета шрифта (от 1 до 8)"
save -i "путь к файлу" 
     -o "путь к сохранению"
     (сохраняет png файл)
exit
```

# Пример

```
image -w 2000 -h 2000
logic -r 1 -g 2
color -b 2 -f 5
save
exit
```
